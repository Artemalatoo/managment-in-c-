import java.util.*;
public class Student {
    String name;
    int id;
    String major;
    int gpa;
static void displayInfo(){
    System.out.println("Study well");
}

    id=i;
    major=m;

}

        Student s=new Student("Artem",230106020,"IT",4);
        System.out.println(s.name+" "+s.id+" "+s.major+" "+s.gpa);
        displayInfo();
    }
}
